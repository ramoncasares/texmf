\MT:
\MT: def mode_setup = relax enddef;
\MT: input graphbase.mf;
\MT:

\def\mfpicopenenv{\MTline{ mfpicenv;}%
 \MTline{ unitlen := 1.0pt\#;}%
 \MTline{ xneg := 0; yneg := 0;}%
 \MTline{ xscale := 1; yscale := 1;}%
 \MTline{ interim hdwdr:=1; interim hdten:=1;}%
 \MTline{ interim penwd:=0.5 pt;}%
 \MTline{}}

\def\mfpicbeginchar(#1,#2,#3);{%
 \MTline{ xpos := #1; ypos := #2;}%
 \MTbeginchar(#1,#2,#3);%
 \MTline{ setztr; pickup pencircle scaled penwd;}}

\def\mfpicendchar;{\MTendchar;}

\def\mfpiccloseenv{\MTline{endmfpicenv;}\MTline{}}
